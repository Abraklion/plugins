Это содержание моей всплывающей подсказки!
Код: 
$('.js-tooltip-7').tooltipster({
    contentCloning : true
})

Параметры:

Код: 
$('.js-tooltip-8').tooltipster({
  debug: true, //-> отладка.

  trigger: 'click', //-> событие открывающие подсказку

  animation: 'grow', //-> анимация.
  animationDuration: [350,350], //-> скорость анимации.
  delay: [300,300], //-> задержка перед.
  delayTouch: [300,500], //-> задержка перед с сенсора.

  updateAnimation: 'rotate', //-> анимация при изменения content
  timer: 0, //-> существования подсказки перед закрытием.

  theme: 'tooltipster-punk',  //-> тема.

  content: null, //-> задать текст подсказки параметром.
  contentAsHTML: false, //-> если не надо экранировать HTML в content.
  contentCloning: false, //-> клонирования элемента посказки.

  restoration: 'none', //-> восстанавливает атрибут TITLE.
  selfDestruction: true, //-> должна ли подсказка самоуничтожаться из DOM.

  interactive: true, //-> взаимодействовать с содержимым подсказки.
  multiple: false, //-> разместить несколько подсказок на одном элементе.

  side: 'top', //-> сторона всплывающей подсказки.
  maxWidth: null, //-> максимальная ширина подсказки.
  minWidth: 0, //-> минимальную ширину подсказки.
  distance: 6, //-> расстояние между элементом и подсказкой.
  repositionOnScroll: false,//-> перемещает всплывающую подсказку.
  arrow: true, //-> стрелка к подсказки.
  zIndex: 9999999, //-> позиц. по оси Z.

  triggerClose: {}, //-> переопредилить события открытия подсказки.
  triggerOpen: {}, //-> переопредилить события закрытия подсказки.

  viewportAware: true, //-> полностью видна на экране при открытии.

  functionInit: null, //-> запускаться при создании экземпляра.
  functionBefore: null, //-> запускаться до открытия подсказки.
  functionReady: null, //-> запускаться когда подсказка добавлена в DOM.
  functionAfter: null, //-> запускаться подсказка закрыта и удалена из DOM.
  functionFormat: null, //-> форматировать content.
  functionPosition: null, //-> запускаться изменении положения подсказки.
})

Методы:

Методы "open" открывает подсказку:
Код: 
let $tooltip9 = $('.js-tooltip-9')

$tooltip9
  .tooltipster()
  .tooltipster('open')
Методы "close" закрывает подсказку:
Код: 
let $tooltip9 = $('.js-tooltip-9')

$tooltip9
  .tooltipster()
  .tooltipster('open')
Методы "content(сеттер)" / "content(геттер)" обновляет содержимое всплывающей подсказки:
Код: 
let $tooltip11 = $('.js-tooltip-11')

$tooltip11.tooltipster()

$tooltip11
  .tooltipster('content','Новое содержимое подсказки')

console.log($tooltip11.tooltipster('content'))
Методы "destroy" закрывает и уничтожает функциональность всплывающей подсказки:
Код: 
let $tooltip12 = $('.js-tooltip-12')

$tooltip12.tooltipster()

$('.js-btn-2').on('click', () => {
  $tooltip12.tooltipster('destroy')
})
Методы "disable/enable" отключить и включить всплывающию подсказку:
Код: 
let $tooltip13 = $('.js-tooltip-13')

$tooltip13.tooltipster()

$('.js-btn-3').on('click', function () {

  let $this = $(this)

  if($this.text() === 'Отключить') {
    $tooltip13.tooltipster('disable')
    $this.text('Включить')
  } else {
    $tooltip13.tooltipster('enable')
    $this.text('Отключить')
  }

})
